function main(splash, args)
  assert(splash:go(args.url))
  assert(splash:wait(0.5))
  splash.private_mode_enabled = false
  LTC = splash::select("#marketOverviewContainer > section > div > article > div > div.filterPanel___2zFYQ > div:nth-child(3)")
  LTC::mouse_click()
  assert(splash:wait(0.5))
  return {
    png = splash:png(),
  }
end
